\def\sigmoidfigure
{
    \begin{tikzpicture}[trim axis left, trim axis right, baseline]
        \begin{axis}[axis lines=none,width=2cm,height=2cm,
                ylabel shift=-0.1cm]
            \addplot[color=black]{1/(1+exp(-x))-2};
        \end{axis}
    \end{tikzpicture}
}

\def\tanhfigure
{
    \begin{tikzpicture}[trim axis left, trim axis right, baseline]
        \begin{axis}[axis lines=none,width=2cm,height=2cm,
                ylabel shift=-0.1cm]
            \addplot[color=black]{(exp(2*x) -1)/(exp(2*x) + 1)};
        \end{axis}
    \end{tikzpicture}
}
